export class StackPillComponent {
  tag = input.required();
  category = input.required();
  isFlipped = signal(false);
  pillConfig = computed(() => {
    const tagName = this.tag(); // Invocação correta do Signal
    const config = TAG_CONFIG[tagName] ?? {
      label: tagName,
      iconPath: `assets/icons/${tagName.toLowerCase()}.svg`,
      category: 'frontend' as const
    };
    return config;
  });
  flipPill() {
    this.isFlipped.update(flipped => !flipped);
  }
}
